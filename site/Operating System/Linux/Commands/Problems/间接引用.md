```sh
# params:

deploy_127_0_0_1="true"

# script:

host=127_0_0_1

deploy=deploy_$host
echo '$deploy': $deploy
echo '$deploy_127_0_0_1': $deploy_127_0_0_1
echo '\$$deploy': \$$deploy
echo '$(eval echo \$$deploy)': $(eval echo \$$deploy)
echo "\`eval echo \'\$\'\$deploy\`": `eval echo '$'$deploy`
echo '${!deploy}': ${!deploy}

if [ ${!deploy} = "false" ]; then exit 0; fi

echo start deploying: $host
```
